4. Gesti√≥n de Estado en React

Manejar datos din√°micos en la aplicaci√≥n

¬øQu√© es el Estado?

El estado es un objeto que contiene datos que cambian con el tiempo y afectan lo que se renderiza en pantalla.

Estado vs Props:

  • Props: Datos que vienen del padre (read-only, no se modifican)
  • Estado: Datos internos del componente (mutable, se puede modificar)

Cu√°ndo necesitas estado:

  • ‚úì Formularios (inputs controlados)
  • ‚úì Toggles (mostrar/ocultar, men√∫s)
  • ‚úì Datos de APIs (usuarios, productos)
  • ‚úì Contadores, timers
  • ‚úì Informaci√≥n del usuario (autenticaci√≥n, tema)

Tipos de estado:

1. Estado Local    ‚Üí useState en un componente
2. Estado Elevado  ‚Üí Compartido entre componentes hermanos
3. Estado Global   ‚Üí useContext, Redux, Zustand
4. Estado Servidor ‚Üí React Query, SWR

Estado Local

El estado local pertenece a un solo componente y se maneja con useState.

function ListaTareas() {
  const [tareas, setTareas] = useState([]);
  const [nuevaTarea, setNuevaTarea] = useState('');
  
  const agregarTarea = () => {
    if (nuevaTarea.trim() === '') return;
    
    const tarea = {
      id: Date.now(),
      texto: nuevaTarea,
      completada: false
    };
    
    setTareas([...tareas, tarea]);
    setNuevaTarea('');
  };
  
  const eliminarTarea = (id) => {
    setTareas(tareas.filter(t => t.id !== id));
  };
  
  const toggleCompletada = (id) => {
    setTareas(tareas.map(t => 
      t.id === id ? { ...t, completada: !t.completada } : t
    ));
  };
  
  return (
    <div>
      <input 
        value={nuevaTarea}
        onChange={(e) => setNuevaTarea(e.target.value)}
        placeholder="Nueva tarea"
      />
      <button onClick={agregarTarea}>Agregar</button>
      
      <ul>
        {tareas.map(tarea => (
          <li key={tarea.id}>
            <input 
              type="checkbox"
              checked={tarea.completada}
              onChange={() => toggleCompletada(tarea.id)}
            />
            <span style={{ 
              textDecoration: tarea.completada ? 'line-through' : 'none' 
            }}>
              {tarea.texto}
            </span>
            <button onClick={() => eliminarTarea(tarea.id)}>
              Eliminar
            </button>
          </li>
        ))}
      </ul>
    </div>
  );
}

Context API - Estado Global

Context API evita prop drilling (pasar props a trav√©s de muchos niveles). Ideal para datos globales: tema, idioma, autenticaci√≥n.

Ejemplo completo: Tema

// 1. Crear contexto
import { createContext, useContext, useState } from 'react';

const ThemeContext = createContext();

// 2. Provider personalizado
export function ThemeProvider({ children }) {
  const [theme, setTheme] = useState('light');
  
  const toggleTheme = () => {
    setTheme(prevTheme => prevTheme === 'light' ? 'dark' : 'light');
  };
  
  const value = { theme, toggleTheme };
  
  return (
    <ThemeContext.Provider value={value}>
      {children}
    </ThemeContext.Provider>
  );
}

// 3. Custom hook para consumir
export function useTheme() {
  const context = useContext(ThemeContext);
  if (!context) {
    throw new Error('useTheme debe usarse dentro de ThemeProvider');
  }
  return context;
}

// 4. App.js
import { ThemeProvider } from './context/ThemeContext';

function App() {
  return (
    <ThemeProvider>
      <Header />
      <Main />
      <Footer />
    </ThemeProvider>
  );
}

// 5. Usar en cualquier componente (sin prop drilling!)
import { useTheme } from './context/ThemeContext';

function Header() {
  const { theme, toggleTheme } = useTheme();
  
  return (
    <header className={theme}>
      <h1>Mi App</h1>
      <button onClick={toggleTheme}>
        Cambiar a {theme === 'light' ? 'dark' : 'light'}
      </button>
    </header>
  );
}

function Main() {
  const { theme } = useTheme();
  
  return (
    <main className={theme}>
      <p>Tema actual: {theme}</p>
    </main>
  );
}

// CartContext.js
const CartContext = createContext();

export function CartProvider({ children }) {
  const [items, setItems] = useState([]);
  
  const addItem = (producto) => {
    setItems(prevItems => {
      const existe = prevItems.find(i => i.id === producto.id);
      if (existe) {
        return prevItems.map(i =>
          i.id === producto.id ? { ...i, cantidad: i.cantidad + 1 } : i
        );
      }
      return [...prevItems, { ...producto, cantidad: 1 }];
    });
  };
  
  const removeItem = (id) => {
    setItems(prevItems => prevItems.filter(i => i.id !== id));
  };
  
  const total = items.reduce((sum, i) => sum + (i.precio * i.cantidad), 0);
  
  return (
    <CartContext.Provider value={{ items, addItem, removeItem, total }}>
      {children}
    </CartContext.Provider>
  );
}

export const useCart = () => useContext(CartContext);

// ProductCard.js
function ProductCard({ producto }) {
  const { addItem } = useCart();
  
  return (
    <div>
      <h3>{producto.nombre}</h3>
      <p>${producto.precio}</p>
      <button onClick={() => addItem(producto)}>
        Agregar al carrito
      </button>
    </div>
  );
}

useReducer - Estado Complejo

useReducer es como useState pero para l√≥gica de estado compleja. Similar a Redux.

Cu√°ndo usar useReducer:

  • ‚úì Estado con m√∫ltiples sub-valores relacionados
  • ‚úì L√≥gica compleja de actualizaci√≥n
  • ‚úì Siguiente estado depende del anterior
  • ‚úì Quieres separar l√≥gica de UI

Estructura b√°sica:

import { useReducer } from 'react';

// 1. Definir reducer (funci√≥n pura que calcula nuevo estado)
function reducer(state, action) {
  switch (action.type) {
    case 'INCREMENT':
      return { count: state.count + 1 };
    case 'DECREMENT':
      return { count: state.count - 1 };
    case 'RESET':
      return { count: 0 };
    default:
      return state;
  }
}

// 2. Usar en componente
function Counter() {
  const [state, dispatch] = useReducer(reducer, { count: 0 });
  
  return (
    <div>
      <p>Count: {state.count}</p>
      <button onClick={() => dispatch({ type: 'INCREMENT' })}>+</button>
      <button onClick={() => dispatch({ type: 'DECREMENT' })}>-</button>
      <button onClick={() => dispatch({ type: 'RESET' })}>Reset</button>
    </div>
  );
}

Ejemplo complejo: Formulario con validaci√≥n

const initialState = {
  username: '',
  email: '',
  password: '',
  errors: {},
  isSubmitting: false
};

function formReducer(state, action) {
  switch (action.type) {
    case 'UPDATE_FIELD':
      return {
        ...state,
        [action.field]: action.value,
        errors: { ...state.errors, [action.field]: '' }
      };
    
    case 'SET_ERROR':
      return {
        ...state,
        errors: { ...state.errors, [action.field]: action.error }
      };
    
    case 'SUBMIT_START':
      return { ...state, isSubmitting: true };
    
    case 'SUBMIT_SUCCESS':
      return initialState;
    
    case 'SUBMIT_ERROR':
      return { ...state, isSubmitting: false };
    
    default:
      return state;
  }
}

function RegistroForm() {
  const [state, dispatch] = useReducer(formReducer, initialState);
  
  const handleChange = (field, value) => {
    dispatch({ type: 'UPDATE_FIELD', field, value });
  };
  
  const validate = () => {
    let isValid = true;
    
    if (state.username.length < 3) {
      dispatch({ 
        type: 'SET_ERROR', 
        field: 'username', 
        error: 'M√≠nimo 3 caracteres' 
      });
      isValid = false;
    }
    
    if (!state.email.includes('@')) {
      dispatch({ 
        type: 'SET_ERROR', 
        field: 'email', 
        error: 'Email inv√°lido' 
      });
      isValid = false;
    }
    
    return isValid;
  };
  
  const handleSubmit = async (e) => {
    e.preventDefault();
    
    if (!validate()) return;
    
    dispatch({ type: 'SUBMIT_START' });
    
    try {
      // Simular API call
      await new Promise(resolve => setTimeout(resolve, 1000));
      dispatch({ type: 'SUBMIT_SUCCESS' });
      alert('Registro exitoso!');
    } catch (error) {
      dispatch({ type: 'SUBMIT_ERROR' });
    }
  };
  
  return (
    <form onSubmit={handleSubmit}>
      <input
        value={state.username}
        onChange={(e) => handleChange('username', e.target.value)}
        placeholder="Username"
      />
      {state.errors.username && <span>{state.errors.username}</span>}
      
      <input
        value={state.email}
        onChange={(e) => handleChange('email', e.target.value)}
        placeholder="Email"
      />
      {state.errors.email && <span>{state.errors.email}</span>}
      
      <input
        type="password"
        value={state.password}
        onChange={(e) => handleChange('password', e.target.value)}
        placeholder="Password"
      />
      
      <button disabled={state.isSubmitting}>
        {state.isSubmitting ? 'Enviando...' : 'Registrar'}
      </button>
    </form>
  );
}

useReducer + Context (mini Redux):

// TodoContext.js
const TodoContext = createContext();

const todoReducer = (state, action) => {
  switch (action.type) {
    case 'ADD_TODO':
      return [...state, { id: Date.now(), text: action.text, done: false }];
    case 'TOGGLE_TODO':
      return state.map(todo =>
        todo.id === action.id ? { ...todo, done: !todo.done } : todo
      );
    case 'DELETE_TODO':
      return state.filter(todo => todo.id !== action.id);
    default:
      return state;
  }
};

export function TodoProvider({ children }) {
  const [todos, dispatch] = useReducer(todoReducer, []);
  
  return (
    <TodoContext.Provider value={{ todos, dispatch }}>
      {children}
    </TodoContext.Provider>
  );
}

export const useTodos = () => useContext(TodoContext);

// Componente
function TodoList() {
  const { todos, dispatch } = useTodos();
  const [input, setInput] = useState('');
  
  const addTodo = () => {
    dispatch({ type: 'ADD_TODO', text: input });
    setInput('');
  };
  
  return (
    <div>
      <input value={input} onChange={(e) => setInput(e.target.value)} />
      <button onClick={addTodo}>Agregar</button>
      {todos.map(todo => (
        <div key={todo.id}>
          <input
            type="checkbox"
            checked={todo.done}
            onChange={() => dispatch({ type: 'TOGGLE_TODO', id: todo.id })}
          />
          <span>{todo.text}</span>
          <button onClick={() => dispatch({ type: 'DELETE_TODO', id: todo.id })}>
            X
          </button>
        </div>
      ))}
    </div>
  );
}
