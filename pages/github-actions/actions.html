Actions Populares

# Checkout del código
- uses: actions/checkout@v3

# Setup Node.js
- uses: actions/setup-node@v3
  with:
    node-version: 18

# Setup Python
- uses: actions/setup-python@v4
  with:
    python-version: '3.11'

# Cache dependencies
- uses: actions/cache@v3
  with:
    path: node_modules
    key: ${{ runner.os }}-node-${{ hashFiles('**/package-lock.json') }}

# Deploy to GitHub Pages
- uses: peaceiris/actions-gh-pages@v3
  with:
    github_token: ${{ secrets.GITHUB_TOKEN }}
    publish_dir: ./dist

Custom Action

# action.yml
name: 'My Action'
description: 'Does something'
inputs:
  name:
    description: 'Name'
    required: true
runs:
  using: 'node16'
  main: 'index.js'

Composite Action

# action.yml
name: 'Setup Project'
runs:
  using: 'composite'
  steps:
    - run: npm install
      shell: bash
    - run: npm run build
      shell: bash
