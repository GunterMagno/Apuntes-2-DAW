2. Workflows

Archivos YAML de configuración

Estructura Básica

name: CI

on: [push, pull_request]

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Run tests
        run: npm test

Ubicación: .github/workflows/ci.yml

Triggers (on)

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]
  schedule:
    - cron: '0 0 * * *'  # Diario a medianoche
  workflow_dispatch:  # Manual

Jobs

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - run: npm test
  
  deploy:
    needs: test  # Espera a 'test'
    runs-on: ubuntu-latest
    steps:
      - run: npm run deploy

Steps

steps:
  - uses: actions/checkout@v3
  
  - name: Setup Node
    uses: actions/setup-node@v3
    with:
      node-version: 18
  
  - run: npm install
  - run: npm test
