3. Servidor HTTP

Crear servidores web con Node.js

Crear un Servidor HTTP

const http = require('http');

const server = http.createServer((req, res) => {
    res.writeHead(200, { 'Content-Type': 'text/plain' });
    res.end('Hola Mundo');
});

server.listen(3000, () => {
    console.log('Servidor corriendo en http://localhost:3000');
});

Request y Response

req (request): Objeto con información de la petición

const server = http.createServer((req, res) => {
    console.log(req.url);        // URL solicitada

    console.log(req.headers);    // Cabeceras HTTP
    
    res.writeHead(200, { 'Content-Type': 'text/html' });

Hola

Routing Básico

const server = http.createServer((req, res) => {
    if (req.url === '/') {
        res.writeHead(200, { 'Content-Type': 'text/html' });
        res.end('
Página de Inicio
');
    } else if (req.url === '/about') {
        res.writeHead(200, { 'Content-Type': 'text/html' });
        res.end('
Acerca de
');
    } else {
        res.writeHead(404, { 'Content-Type': 'text/html' });
        res.end('
404 - Página no encontrada
');
    }
});

Métodos HTTP

const server = http.createServer((req, res) => {
    if (req.method === 'GET' && req.url === '/api/users') {
        res.writeHead(200, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({ users: ['Ana', 'Luis'] }));
    } else if (req.method === 'POST' && req.url === '/api/users') {
        let body = '';
        req.on('data', chunk => {
            body += chunk.toString();
        });
        req.on('end', () => {
            console.log('Datos recibidos:', body);
            res.end('Usuario creado');
        });
    }
});

Responder con JSON

const server = http.createServer((req, res) => {
    const data = {
        message: 'Hola',
        timestamp: new Date(),
        users: ['Ana', 'Luis', 'María']
    };
    
    res.writeHead(200, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify(data));
});

server.listen(3000);
