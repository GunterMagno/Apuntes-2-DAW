5. Asincronía

Callbacks, Promises y Async/Await

Callbacks

Un callback es una función que se pasa como argumento y se ejecuta después de completar una operación asíncrona.

const fs = require('fs');

fs.readFile('archivo.txt', 'utf8', (err, data) => {
    if (err) {
        console.error('Error:', err);
        return;
    }
    console.log(data);
});

Callback Hell: Callbacks anidados difíciles de leer

fs.readFile('1.txt', (err, data1) => {
    fs.readFile('2.txt', (err, data2) => {
        fs.readFile('3.txt', (err, data3) => {
            // Pirámide del doom
        });
    });
});

Promises

Una Promise representa un valor que puede estar disponible ahora, en el futuro, o nunca.

const fs = require('fs/promises');

fs.readFile('archivo.txt', 'utf8')
    .then(data => {
        console.log(data);
        return fs.readFile('otro.txt', 'utf8');
    })
    .then(data2 => {
        console.log(data2);
    })
    .catch(err => {
        console.error('Error:', err);
    });

Crear una Promise

function esperar(ms) {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            resolve('Completado');
        }, ms);
    });
}

esperar(2000).then(msg => console.log(msg));  // Después de 2 segundos

Async/Await

Sintaxis más limpia para trabajar con Promises.

const fs = require('fs/promises');

async function leerArchivos() {
    try {
        const data1 = await fs.readFile('1.txt', 'utf8');
        console.log(data1);
        
        const data2 = await fs.readFile('2.txt', 'utf8');
        console.log(data2);
        
        const data3 = await fs.readFile('3.txt', 'utf8');
        console.log(data3);
    } catch (err) {
        console.error('Error:', err);
    }
}

leerArchivos();

Manejo de Errores

Con callbacks:

fs.readFile('archivo.txt', (err, data) => {
    if (err) {
        // Manejar error
        return;
    }
    // Usar data
});

Con async/await:

async function main() {
    try {
        const data = await fs.readFile('archivo.txt', 'utf8');
        console.log(data);
    } catch (err) {
        console.error('Error:', err.message);
    }
}

Ejemplos Prácticos

Múltiples operaciones en paralelo

async function leerTodos() {
    const [data1, data2, data3] = await Promise.all([
        fs.readFile('1.txt', 'utf8'),
        fs.readFile('2.txt', 'utf8'),
        fs.readFile('3.txt', 'utf8')
    ]);
    console.log(data1, data2, data3);
}
