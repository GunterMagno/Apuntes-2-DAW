4. File System (fs)

Trabajar con archivos y directorios

Leer Archivos

const fs = require('fs');

// Leer archivo (asíncrono)
fs.readFile('archivo.txt', 'utf8', (err, data) => {
    if (err) {
        console.error('Error:', err);
        return;
    }
    console.log(data);
});

// Leer archivo (síncrono)
try {
    const data = fs.readFileSync('archivo.txt', 'utf8');
    console.log(data);
} catch (err) {
    console.error('Error:', err);
}

Escribir Archivos

// Escribir archivo (reemplaza contenido)
fs.writeFile('nuevo.txt', 'Hola Mundo', (err) => {
    if (err) throw err;
    console.log('Archivo creado');
});

// Añadir al final del archivo
fs.appendFile('archivo.txt', '\nNueva línea', (err) => {
    if (err) throw err;
    console.log('Texto añadido');
});

Síncronas vs Asíncronas

Síncronas (Sync): Bloquean el código hasta terminar

console.log('Inicio');
const data = fs.readFileSync('archivo.txt', 'utf8');
console.log(data);
console.log('Fin');  // Espera a que termine la lectura

Asíncronas: No bloquean el código

console.log('Inicio');
fs.readFile('archivo.txt', 'utf8', (err, data) => {
    console.log(data);
});
console.log('Fin');  // Se ejecuta antes que el console.log de data

fs/promises

const fs = require('fs/promises');

// Con Promises
fs.readFile('archivo.txt', 'utf8')
    .then(data => console.log(data))
    .catch(err => console.error(err));

// Con async/await
async function leerArchivo() {
    try {
        const data = await fs.readFile('archivo.txt', 'utf8');
        console.log(data);
    } catch (err) {
        console.error(err);
    }
}
leerArchivo();

Trabajar con Directorios

const fs = require('fs');

// Crear directorio
fs.mkdir('nueva-carpeta', (err) => {
    if (err) throw err;
    console.log('Carpeta creada');
});

// Leer directorio
fs.readdir('./carpeta', (err, files) => {
    if (err) throw err;
    console.log(files);  // Array de nombres de archivos
});

// Eliminar directorio
fs.rmdir('carpeta', (err) => {
    if (err) throw err;
    console.log('Carpeta eliminada');
});

// Eliminar archivo
fs.unlink('archivo.txt', (err) => {
    if (err) throw err;
    console.log('Archivo eliminado');
});
