Sistema de Módulos en Sass

Módulos en Sass

El sistema de módulos de Sass permite organizar el código en archivos separados y reutilizables.

// _variables.scss
$primary-color: #2965f1;

// main.scss
@use 'variables';

body {
    color: variables.$primary-color;
}

@use

// _colors.scss
$primary: #2965f1;
$secondary: #cd6799;

// _mixins.scss
@mixin flex-center {
    display: flex;
    justify-content: center;
    align-items: center;
}

// styles.scss
@use 'colors';
@use 'mixins';

.container {
    background: colors.$primary;
    @include mixins.flex-center;
}

@forward

// _index.scss (barrel file)
@forward 'variables';
@forward 'mixins';
@forward 'functions';

// main.scss
@use 'index' as *;

body {
    color: $primary-color; // De variables
    @include flex-center;  // De mixins
}
